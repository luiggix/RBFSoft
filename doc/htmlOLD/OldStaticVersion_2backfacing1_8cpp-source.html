00483 void streamLaplace(Mat& G, const Vec& x, const Vec& y, RBF rbf,
00484                    RBF_2DX rbf_2dx, RBF_2DY rbf_2dy)
00485 {
00486 // ----- Wl matrix

00488         for(int i = 1; i <= NI; ++i) {
00489             G(i,j) = rbf_2dx ( x(i), y(i), x(j), y(j) ) + 
00490               rbf_2dy ( x(i), y(i), x(j), y(j) );
00491         }
00492 
00493 // ----- Wb matrix
00494     for(int j = 1; j <= N; ++j)
