00566 void streamLaplace(Mat& G, const Vec& x, const Vec& y,

00573                 dt * ( rbf_2dx ( x(i), y(i), x(j), y(j) ) +

00703 void vorticityBC(Vec& b, Mat& vgrid, const Mat& sgrid, int* point, prec_t beta, Vec& y, prec_t Re)
00704 {
00705     b = 0;  // Source term
00706 
00707 // Edges 1: Backward facing step - Inlet
00708 
00709 /* */
00710     prec_t y_coord;
00711     for(int i = point[0], k = 2; i <= point[1]; ++i, ++k) {
00712         y_coord = y(i) - 0.5;
00713 //      b(i) = 6 * y_coord;

00715 /* *
00716 // Second order approx. (Jensen formula):
00717         b(i) = -(4 * sgrid(2,k) - 0.5 * sgrid(3,k) - 3.5 * sgrid(1,k)) 
00718             / (dx * dx);
00719 /* */
00720     }
00721     
00722 // Edge 2: Outlet
00723     for(int i = point[2], k = 2; i <= point[3]; ++i, ++k) {
00724 /* */
00725         y_coord = y(i) - 0.5;
00726 //      b(i) = 6 * y_coord;
00727         b(i) = Re * 8 * y_coord;
00728 /* *
00729 // Second order approx. (Jensen formula):
00730         b(i) = -(4 * sgrid(Nx-1,k) - 0.5 * sgrid(Nx-2,k) - 3.5 * sgrid(Nx,k)) 
00731             / (dx * dx);
00732 /* */
00733     }   
00734 
00735 // Edges 3: Dirichlet
00736     for(int i = point[4], k = 2; i <= point[5]; ++i, ++k) {
00737 // Second order approx. (Jensen formula):
00738         b(i) = -(4 * sgrid(k, 2) - 0.5 * sgrid(k, 3) - 3.5 * sgrid(k,1) ) 
00739             / (dy * dy);
00740     }
00741     
00742 // Edges 4: Dirichlet
00743     for(int i = point[6], k = 2; i <= point[7]; ++i, ++k) {
00744 // Second order approx. (Jensen formula):
00745         b(i) = -(4 * sgrid(k, Ny-1) - 0.5 * sgrid(k, Ny-2) - 3.5 * sgrid(k,Ny))
00746             / (dy*dy);
00747     }
00748 // Corners:
00749 // Average between wall neighbors:
00750 /* *
00751    b(N-3) = 0.5 * ( b(point[0]) + b(point[4]) );
00752    b(N-2) = 0.5 * ( b(point[1]) + b(point[6]) );
00753    b(N-1) = 0.5 * ( b(point[2]) + b(point[5]) );
00754    b(N  ) = 0.5 * ( b(point[3]) + b(point[7]) );
00755 // Values from analytical parabolic profile
00756 /* */
00757     b(N-3) = -Re * 4;
00758     b(N-2) = Re * 4;
00759     b(N-1) = -Re * 4;
00760     b(N  ) = Re * 4;
00761 /* */
00762 
00763 // ----- Under-relaxation on the vorticity boundary conditions
00764     for(int i = point[0], k = 2; i <= point[1]; ++i, ++k) {
00765         b(i) = beta * b(i) + (1 - beta) * vgrid(1, k);
00766         vgrid(1, k) = b(i);

00771     }

00773         b(i) = beta * b(i) + (1 - beta) * vgrid(k, 1);
00774         vgrid(k, 1) = b(i);
00775     }
00776     for(int i = point[6], k = 2; i <= point[7]; ++i, ++k) {
00777         b(i) = beta * b(i) + (1 - beta) * vgrid(k, Ny);
00778         vgrid(k, Ny) = b(i);    
00779     }
00780 // Corners: under-relaxation
00781     b(N-3) = beta * b(N-3) + (1 - beta) * vgrid(1, 1); 
00782     b(N-2) = beta * b(N-2) + (1 - beta) * vgrid(1,Ny); 
00783     b(N-1) = beta * b(N-1) + (1 - beta) * vgrid(Nx,1);
