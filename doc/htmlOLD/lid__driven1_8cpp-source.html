00342 template<typename RBF, typename RBF_2DX, typename RBF_2DY>
00343 void streamLaplace(Mat& G, const Vec& x, const Vec& y, 
00344                    RBF rbf, RBF_2DX rbf_2dx, RBF_2DY rbf_2dy)
00345 {
00346 // ----- Wl matrix
00347     for(int j = 1; j <= N; ++j) 
00348         for(int i = 1; i <= NI; ++i) {
00349             G(i,j) = rbf_2dx ( x(i), y(i), x(j), y(j) ) +
00350                      rbf_2dy ( x(i), y(i), x(j), y(j) );
00351         }
00352     
00353 // ----- Wb matrix
00354     for(int j = 1; j <= N; ++j) 
00355         for(int i = NI+1; i <= N; ++i)
00356             G(i,j) = rbf ( x(i), y(i), x(j), y(j) );   
00357 }

00362          typename RBF_1DX, typename RBF_1DY,
00363          typename RBF_2DX, typename RBF_2DY>
00364 void vortEquation(Mat& G, const Vec& x, const Vec& y,
00365                   RBF rbf, 
00366                   RBF_1DX rbf_1dx, RBF_1DY rbf_1dy, 
00367                   RBF_2DX rbf_2dx, RBF_2DY rbf_2dy, 
00368                   const Vec& u, const Vec& v, prec_t Re)
00369 {
00370 // ----- WL matrix
00371     for(int j = 1; j <= N; ++j) 
00372         for(int i = 1; i <= NI; ++i) {
00373             G(i,j) = rbf_2dx ( x(i), y(i), x(j), y(j) ) + 
00374                 rbf_2dy ( x(i), y(i), x(j), y(j) ) -
00375                 Re * ( u(i) * rbf_1dx( x(i), y(i), x(j), y(j) ) +
00376                        v(i) * rbf_1dy( x(i), y(i), x(j), y(j) ) );
00377         }
00378 
00379 // ----- WB matrix
00380     for(int j = 1; j <= N; ++j) 
00381         for(int i = NI+1; i <= N; ++i)
00382             G(i,j) = rbf ( x(i), y(i), x(j), y(j) );

00588             0.5 * vgrid(Nx,Ny-1) + 
00589             0.25 * vgrid(Nx-1,Ny-1) );

00592 // ----- Under-relaxation on the vorticity boundary conditions
00593     for(int i = point[0], k = 2; i <= point[1]; ++i, ++k) {
00594         b(i) = beta * b(i) + (1 - beta) * vgrid(1, k);
00595         vgrid(1, k) = b(i);

00597     for(int i = point[2], k = 2; i <= point[3]; ++i, ++k) {
00598         b(i) = beta * b(i) + (1 - beta) * vgrid(Nx, k);
00599         vgrid(Nx, k) = b(i);
00600     }
00601     for(int i = point[4], k = 2; i <= point[5]; ++i, ++k) {
00602         b(i) = beta * b(i) + (1 - beta) * vgrid(k, 1);
00603         vgrid(k, 1) = b(i);
00604     }
00605     for(int i = point[6], k = 2; i <= point[7]; ++i, ++k) {
00606         b(i) = beta * b(i) + (1 - beta) * vgrid(k, Ny);
00607         vgrid(k, Ny) = b(i);    
00608     }
00609 // Corners: under-relaxation
00610     b(N-3) = beta * b(N-3) + (1 - beta) * vgrid(1, 1); 
00611     b(N-2) = beta * b(N-2) + (1 - beta) * vgrid(1,Ny); 
00612     b(N-1) = beta * b(N-1) + (1 - beta) * vgrid(Nx,1); 
00613     b(N  ) = beta * b(N  ) + (1 - beta) * vgrid(Nx,Ny);
00614     vgrid(1 , 1 ) = b(N-3);
00615     vgrid(1 , Ny) = b(N-2);
00616     vgrid(Nx, 1 ) = b(N-1);
00617     vgrid(Nx, Ny) = b(N  );
