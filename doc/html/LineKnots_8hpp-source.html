00111 template<typename prec_t>
00112 bool LineKnots<prec_t>::constructKnots() 
00113 {
00114     prec_t dx = lx / (Nx - 1);
00115 
00116     int index = 0;
00117 // ----- Interior points.
00118     for (int i = 1; i < Nx; ++i, ++index) 
00119         xyz[index].coord[0] = i * dx; 
00120 
00121 // ----- Boundary points
00122     xyz[Nx-2].coord[0] = 0;
00123     xyz[Nx-1].coord[0] = lx;
00124 
00125     return 0;

00138 {
00139     int sp[2], side_x = Nx - 2;

00141     sp[0] = N-2;      // 1

00144 // Add 1 to sp[] to be in agreement with P() matrix indices.
00145     for(int i = 0; i < 2; ++i) sp[i] += 1;
00146 
00147     int bi = P.firstRow();
00148     int ei = P.lastRow();
00149     
00150     if ( nsp ) {
00151         
00152         for(int i = bi; i <= ei; ++i) {
00153             if ( P(i,j) == sp[0] ) sp[0] = -1; 
00154             if ( P(i,j) == sp[1] ) sp[1] = -1;
00155         }
00156         
00157         ei -= 1;
00158 
00159         if (sp[0] != -1) P(ei,j) = sp[0];
00160         else P(ei,j) = N - 2;
00161         
00162         if (sp[1] != -1) P(ei+1,j) = sp[1];
00163         else P(ei+1,j) = 2;
00164     }
