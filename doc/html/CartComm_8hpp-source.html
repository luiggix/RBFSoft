00099 CartComm::CartComm(int argc, char **argv, int r)
00100 {
00101     rank = r;    
00102     if (argc >= 3) {
00103         num_proc[0] = std::atoi(argv[2]); // Because the Cartcomm numeration
00104         num_proc[1] = std::atoi(argv[1]); // is "upside down"
00105     } else if (rank == 0) {
00106         std::cerr << "\n\n DD::CartComm : usage: " 
00107                   << argv[0] << " ns_x ns_y" 
00108                   << "\n ns_x : number of subdomains in x-axis"
00109                   << "\n ns_y : number of subdomains in y-axis \n\n";
00110         std::exit(0);
00111     }
00112 
00113     // create mpi_cart:  2-dimensional, non-periodic
00114     periods[0] = false;
00115     periods[1] = false;
00116     comm = MPI::COMM_WORLD.Create_cart(2, num_proc, periods, true);
00117     
00118     // find your position in the cart
00119     comm.Get_coords(rank, 2, coords);
00120 
00121     // find your neighbors up, down, left and right,
00122     // NOTE: if there is not neigbor in some direction, the result is -1.
00123     comm.Shift(0, 1, neighbors[UP],  neighbors[DOWN]);
00124     comm.Shift(1, 1, neighbors[LEFT], neighbors[RIGHT]);
00125 }
