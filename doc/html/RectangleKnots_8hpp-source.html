00180 {
00181     prec_t dx = lx / (Nx - 1);
00182     prec_t dy = ly / (Ny - 1);
00183     prec_t x0, x1, y0, y1;
00184     prec_t ep = dx > dy ? dy * 0.1 : dx * 0.1;
00185     
00186 // ----- Interior points.
00187     int ei = Nx - 2, ej = Ny - 2;
00188     
00189     int index = 0; // The xyz is a C-array, so begins in 0.
00190     
00191     switch (knots_dist) 
00192     {
00193         case 0: // Uniform 
00194             for (int j = 0; j < ej; ++j)
00195                 for (int i = 0; i < ei; ++i, ++index) {
00196                     xyz[index].coord[0] = (i+1) * dx; 
00197                     xyz[index].coord[1] = (j+1) * dy;
00198                 }
00199             break;              
00200         case 1: // Random       
00201             srand( time(0) );
00202             for (int j = 0; j < ej ; ++j)
00203                 for (int i = 0; i < ei; ++i, ++index) { 
00204                     xyz[index].coord[0] = ep +
00205                         (1-2*ep) * lx * static_cast<prec_t>(rand())/RAND_MAX; 
00206                     xyz[index].coord[1] = ep +
00207                         (1-2*ep) * ly * static_cast<prec_t>(rand())/RAND_MAX; 
00208                 }
00209             break;
00210         case 2: // Random controlled 
00211             srand( time(0) );
00212             ++ei; ++ej;
00213             for (int j = 0; j < ej; ++j)
00214                 for (int i = 0; i < ei; ++i, ++index) {
00215                     x0 = (i + epru) * dx; x1 = (i + 1 - epru) * dx;// Mapping
00216                     y0 = (j + epru) * dy; y1 = (j + 1 - epru) * dy;// to a cell
00217                     xyz[index].coord[0] = x0 + 
00218                         ( static_cast<prec_t>(rand())/RAND_MAX ) * (x1 - x0);
00219                     xyz[index].coord[1] = y0 +
00220                         ( static_cast<prec_t>(rand())/RAND_MAX ) * (y1 - y0);
00221                 }
00222             break;
00223         default:
00224             break;
00225     }
00226     
00227 // ----- Ghost points    
00228     if ( NG > 0 ) {
00229 // Generation of extra ghost-knots around the domain.
00230 // The corners are avoided.
00231         
00232 // Edge 1: x = 0 - dx
00233         for (int j = 0; j < Ny; ++j, ++index) {
00234             xyz[index].coord[0] = -dx;
00235             xyz[index].coord[1] = j * dy;
00236         }
00237 // Edge 2: x = lx + dx
00238         for (int j = 0; j < Ny; ++j, ++index) {
00239             xyz[index].coord[0] = lx + dx;
00240             xyz[index].coord[1] = j * dy;
00241         }
00242         
00243 // Edge 3: y = 0 - dy
00244         for (int i = 0; i < Nx; ++i, ++index) {
00245             xyz[index].coord[0] = i * dx;
00246             xyz[index].coord[1] = -dy;
00247         }
00248         
00249 // Edge 4: y = ly + dy
00250         for (int i = 0; i < Nx; ++i, ++index) {
00251             xyz[index].coord[0] = i * dx;
00252             xyz[index].coord[1] = ly + dy;
00253         }
00254     }
00255     
00256 // ----- Boundary points
00257 
00258 // Edges 1 and 2: x = constant
00259     for (int i = 0; i <= 1; ++i)
00260         for (int j = 1; j < Ny-1; ++j, ++index) {
00261             xyz[index].coord[0] = lx * i;
00262             xyz[index].coord[1] = j * dy;
00263         }
00264     
00265 // Edges 3 and 4: y = constant
00266     for (int j = 0; j <= 1; ++j)
00267         for (int i = 1; i < Nx-1; ++i, ++index) {
00268             xyz[index].coord[0] = i * dx;
00269             xyz[index].coord[1] = ly * j;
00270         }
00271 
00272 // Corners    
00273     if (corners) {
00274         for (int i = 0; i <= 1; ++i)
00275             for (int j = 0; j <= 1; ++j, ++index) {
00276                 xyz[index].coord[0] = lx * i;
00277                 xyz[index].coord[1] = ly * j;
00278             }
00279     }
