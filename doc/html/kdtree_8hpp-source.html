00617         // ditto for the right

00626             // staying balanced.

00628             // See top for notes.

00630             // compare the children based on this level's criteria...
00631             // (this gives virtually random results)
00632             if (compare(_S_right(node), _S_left(node)))
00633                // the right is smaller, get our replacement from the SMALLEST on the right
00634                candidate = _M_get_j_min(std::pair<_Link_type,size_t>(_S_right(node),level), level+1);

00725       _M_find(_Link_const_type node, const_reference value, size_t const level) const throw ()
00726       {
00727          // be aware! This is very different to normal binary searches, because of the <=
00728          // relationship used. See top for notes.
00729          // Basically we have to check ALL branches, as we may have an identical node
00730          // in different branches.
00731           const_iterator found = this->end();
00732 
00733         _Node_compare compare(level % __K,_M_acc);
00734         if (!compare(node,value))   // note, this is a <= test
00735           {
00736            // this line is the only difference between _M_find_exact() and _M_find()

00738               return const_iterator(node);   // return right away
00739             if (_S_left(node))
00740                found = _M_find(_S_left(node), value, level+1);
00741           }
00742         if ( _S_right(node) && found == this->end() && !compare(value,node))   // note, this is a <= test
00743             found = _M_find(_S_right(node), value, level+1);
00744         return found;

00754           const_iterator found = this->end();

00768             found = _M_find_exact(_S_right(node), value, level+1);

00898        // WARNING: Calculates and RETURNS dist^2 (for speed)
00899        // NOTE: CENTER is a region of zero area.  It is the point we are aiming for.
00900        //
00901        // How it works: Starting with a centerpt (single-pt in a region, with a range
00902        // attached to it), and bounds, it first calculates the distance to THIS node,
00903        // and adjusts the center's range DOWN if its closer.  No point looking further
00904        // than it needs to look.  A form of a dynamic find_within_range.
00905        // It expands the bounds as usual and sees if it intersects the centerpt+range.
00906        // And so it goes ...
00907 
00908          // substitude predicate for normal find_nearest()s
00909          struct always_true
00910          {
00911             bool operator()( _Val const& ) const { return true; }
00912          };
00913 
00914          template <class Predicate>
00915          std::pair<const_iterator,distance_type>
00916         _M_find_nearest( _Link_const_type __N, typename _Region::_CenterPt __CENTER,
00917                              _Region const& __BOUNDS,
00918                              size_t const __L,
00919                              Predicate predicate ) const throw ()
00920         {
00921            std::pair<const_iterator,distance_type> best(this->end(),__CENTER.second);
00922 
00923            // we ignore this node if the predicate isn't true
00924            if (predicate(*const_iterator(__N)))
00925            {
00926               distance_type dist = 0;
00927               for ( size_t i = 0; i != __K; ++i )
00928               {
00929                  dist += _M_square( __CENTER.first._M_low_bounds[i] - _M_acc(_S_value(__N),i) );
00930               }
00931 #ifdef KDTREE_CHECK_PERFORMANCE
00932               ++num_dist_calcs;
00933 #endif
00934               dist = sqrt(dist);
00935 
00936               best.first = __N;
00937               best.second = dist;
00938            }
00939 
00940            // adjust our CENTER target
00941            __CENTER.second = std::min(__CENTER.second,best.second);
00942 
00943           if (_S_left(__N))
00944             {
00945               _Region __bounds(__BOUNDS);
00946               __bounds.set_high_bound(_S_value(__N), __L);
00947               if (__bounds.intersects_with(__CENTER))
00948               {
00949                  std::pair<const_iterator,distance_type> left =
00950                     _M_find_nearest( _S_left(__N), __CENTER, __bounds, __L+1, predicate);
00951                  // check if better than what I found
00952                  if (left.second < best.second) best = left;
00953               }
00954             }
00955 
00956            // adjust our center target (only useful if left found something closer)
00957            __CENTER.second = std::min(__CENTER.second,best.second);
00958 
00959           if (_S_right(__N))
00960             {
00961               _Region __bounds(__BOUNDS);
00962               __bounds.set_low_bound(_S_value(__N), __L);
00963               if (__bounds.intersects_with(__CENTER))
00964               {
00965                  std::pair<const_iterator,distance_type> right =
00966                     _M_find_nearest( _S_right(__N), __CENTER, __bounds, __L+1, predicate);
00967                  // check if better than what I found
00968                  if (right.second < best.second) best = right;
00969                }
00970             }
00971 
00972           return best;
