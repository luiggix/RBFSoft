00096 // ----- Get the correct lenght, num. of points, shifts and overlapping of 
00097 // ----- every subdomain, based on its position in the CartComm
00098 // The numeration of CartComm is as in matrices of Linear Algebra
00099 // therefore the I goes in -y direction, while J goes in x direction
00100 // (I,J) \in [0, NP_I - 1] X [0, NP_J - 1]
00101 //

00104         NP_J = cart.getNumProc_J();
00105         I = cart.get_I();
00106         J = cart.get_J();
00107 
00108         prec_t llx = l_x / NP_J; // Size of the subdomain not
00109         prec_t lly = l_y / NP_I; // including the overlapping
00110 
00111         lx = l_x / NP_J + 2 * g_x; // Same size for all subdom, except for
00112         ly = l_y / NP_I + 2 * g_y; // the overlapping.
00113         overlap[0] = g_x;  
00114         overlap[1] = g_y;  
00115         Nx = n_x;        
00116         Ny = n_y;        
00117 
00118 // Redimensioning the lenght depending on the location of the subdomain 
00119         if ( I == 0)         ly -= g_y;
00120         if ( I == NP_I - 1 ) ly -= g_y;
00121         if ( J == 0)         lx -= g_x;
00122         if ( J == NP_J - 1 ) lx -= g_x;

00124         dx = lx / (Nx - 1); 
00125         dy = ly / (Ny - 1); 
00126         shift[0] = llx * J; 
00127         shift[1] = lly * (NP_I - I - 1);
00128 
00129 // Correction of the shift by the overlapping
00130         if ( I != NP_I - 1 ) { shift[1] -= g_y; }
00131         if ( J != 0 ) { shift[0] -= g_x; }

00136         NB = 2 * (Nx + Ny) - 4;
00137

00230 bool OVRectangleKnots<prec_t>::constructKnots() 
00231 {
00232     prec_t x0, x1, y0, y1;
00233     prec_t ep = dx > dy ? dy * 0.1 : dx * 0.1;
00234 
00235 // ----- Interior points.
00236     int ei = Nx - 2, ej = Ny - 2;
00237 
00238     int index = 0; // The xyz is a C-array, so it begins in 0.
00239 
00240     switch (knots_dist) 
00241     {
00242         case 0: // Uniform 
00243             for (int j = 0; j < ej; ++j)
00244                 for (int i = 0; i < ei; ++i, ++index) {
00245                     xyz[index].coord[0] = (i+1) * dx + shift[0]; 
00246                     xyz[index].coord[1] = (j+1) * dy + shift[1];
00247                 }
00248             break;              
00249         case 1: // Random       
00250             srand( time(0) + rank);
00251             for (int j = 0; j < ej ; ++j)
00252                 for (int i = 0; i < ei; ++i, ++index) { 
00253                     xyz[index].coord[0] = ep + shift[0] +
00254                         (1-2*ep) * lx * static_cast<prec_t>(rand())/RAND_MAX; 
00255                     xyz[index].coord[1] = ep + shift[1] +
00256                         (1-2*ep) * ly * static_cast<prec_t>(rand())/RAND_MAX; 
00257                 }
00258             break;
00259         case 2:   
00260             srand( time(0) + rank);
00261             ++ei; ++ej;
00262             for (int j = 0; j < ej; ++j)
00263                 for (int i = 0; i < ei; ++i, ++index) {
00264                     x0 = (i + epru) * dx; x1 = (i + 1 - epru) * dx;// Mapping
00265                     y0 = (j + epru) * dy; y1 = (j + 1 - epru) * dy;// to a cell
00266                     xyz[index].coord[0] = x0 + shift[0] +
00267                         ( static_cast<prec_t>(rand())/RAND_MAX ) * (x1 - x0);
00268                     xyz[index].coord[1] = y0 + shift[1] +
00269                         ( static_cast<prec_t>(rand())/RAND_MAX ) * (y1 - y0);
00270                 }
00271             break;
00272         default:
00273             break;
00274     }
00275 
00276 // ----- Ghost Boundaries or real boundaries, depends on the subdomain
00277 
00278 // Edges 1: x = shift[0]
00279     for (int j = 1; j < Ny - 1; ++j, ++index) {
00280         xyz[index].coord[0] = shift[0];
00281         xyz[index].coord[1] = j * dy + shift[1];
00282     }
00283 
00284 // Edges 2: x = lx + shift[0]
00285     for (int j = 1; j < Ny - 1; ++j, ++index) {
00286         xyz[index].coord[0] = lx + shift[0];
00287         xyz[index].coord[1] = j * dy + shift[1];
00288     }
00289 
00290 // Edges 3: y = shift[1]    
00291     for (int i = 1; i < Nx - 1; ++i, ++index) {
00292         xyz[index].coord[0] = i * dx + shift[0];
00293         xyz[index].coord[1] = shift[1];
00294     }
00295 
00296 // Edges 4: y = ly + shift[1]    
00297     for (int i = 1; i < Nx - 1; ++i, ++index) {
00298         xyz[index].coord[0] = i * dx + shift[0];
00299         xyz[index].coord[1] = ly + shift[1];
00300     }
00301 
00302 // Corners:
00303 // 1
00304     xyz[N-4].coord[0] = shift[0];
00305     xyz[N-4].coord[1] = shift[1];
00306 
00307 // 2
00308     xyz[N-3].coord[0] = shift[0];
00309     xyz[N-3].coord[1] = shift[1] + ly;
00310 
00311 // 3
00312     xyz[N-2].coord[0] = shift[0] + lx;
00313     xyz[N-2].coord[1] = shift[1];
00314 
00315 // 4
00316     xyz[N-1].coord[0] = shift[0] + lx;
00317     xyz[N-1].coord[1] = shift[1] + ly;
00318 
00319     return 0;

00323 inline bool OVRectangleKnots<prec_t>::constructOverlapping(CartComm& cart)
00324 {
00325     MPI::Request req[8];
00326     MPI::Status  sta[8];
00327 
00328     const int *neighbor = cart.getNeighbors();
00329 
00330 // ----- Exchange of the size of overlapping.
00331     prec_t gx_L, gx_R, gy_U, gy_D;
00332     req[0] = cart.comm.Isend(&overlap[0], 1, MPI::DOUBLE, neighbor[LEFT],  1);
00333     req[1] = cart.comm.Isend(&overlap[0], 1, MPI::DOUBLE, neighbor[RIGHT], 1);
00334     req[2] = cart.comm.Isend(&overlap[1], 1, MPI::DOUBLE, neighbor[UP],    1);
00335     req[3] = cart.comm.Isend(&overlap[1], 1, MPI::DOUBLE, neighbor[DOWN],  1);
00336 
00337     req[4] = cart.comm.Irecv(&gx_L, 1, MPI::DOUBLE, neighbor[LEFT],  1);
00338     req[5] = cart.comm.Irecv(&gx_R, 1, MPI::DOUBLE, neighbor[RIGHT], 1);
00339     req[6] = cart.comm.Irecv(&gy_U, 1, MPI::DOUBLE, neighbor[UP],    1);
00340     req[7] = cart.comm.Irecv(&gy_D, 1, MPI::DOUBLE, neighbor[DOWN],  1);
00341 
00342     MPI::Request::Waitall(8, req, sta);    
00343 
00344 // ----- Exchange of the number of points for the overlapping.
00345     int Ny_L, Ny_R, Nx_U, Nx_D;
00346     req[0] = cart.comm.Isend(&Ny, 1, MPI::INT, neighbor[LEFT],  2);
00347     req[1] = cart.comm.Isend(&Ny, 1, MPI::INT, neighbor[RIGHT], 2);
00348     req[2] = cart.comm.Isend(&Nx, 1, MPI::INT, neighbor[UP],    2);
00349     req[3] = cart.comm.Isend(&Nx, 1, MPI::INT, neighbor[DOWN],  2);
00350 
00351     req[4] = cart.comm.Irecv(&Ny_L, 1, MPI::INT, neighbor[LEFT],  2);
00352     req[5] = cart.comm.Irecv(&Ny_R, 1, MPI::INT, neighbor[RIGHT], 2);
00353     req[6] = cart.comm.Irecv(&Nx_U, 1, MPI::INT, neighbor[UP],    2);
00354     req[7] = cart.comm.Irecv(&Nx_D, 1, MPI::INT, neighbor[DOWN],  2);
00355 
00356     MPI::Request::Waitall(8, req, sta);
00357 
00358 // ----- Construction of overlapping zones for the neighbor subdomains.
00359 //       array_L, array_R, array_U and array_D will contain the coordinates
00360 //       of the overlapping lines. The last element of these arrays contains
00361 //       the x- or y-coordinate, depending on the orientation of the line.
00362     prec_t dh, xya;
00363     if ( neighbor[LEFT] != -1)  {  // Edge 1
00364         array_L.resize(Ny_L + 1);
00365         xya = shift[0] + overlap[0] + gx_L;
00366         dh = ly / (Ny_L - 1);
00367         for(int i = 1; i <= Ny_L; ++i) 
00368             array_L(i) = (i - 1) * dh + shift[1];
00369         array_L(Ny_L + 1) = xya; // the x-coordinate
00370     }
00371 
00372     if ( neighbor[RIGHT] != -1) {  // Edge 2
00373         array_R.resize(Ny_R + 1);
00374         xya = shift[0] + lx - overlap[0] - gx_R;
00375         dh = ly / (Ny_R - 1);
00376         for(int i = 1; i <= Ny_R; ++i)
00377             array_R(i) = (i - 1) * dh + shift[1];
00378         array_R(Ny_R + 1) = xya; // the x-coordinate
00379     }
00380 
00381     if ( neighbor[UP] != -1)    {  // Edge 3
00382         array_U.resize(Nx_U + 1);
00383         xya = shift[1] + ly - overlap[1] - gy_U;
00384         dh = lx / (Nx_U - 1);
00385         for(int i = 1; i <= Nx_U; ++i)
00386             array_U(i) = (i - 1) * dh + shift[0];
00387         array_U(Nx_U + 1) = xya; // the y-coordinate
00388     }
00389 
00390     if ( neighbor[DOWN] != -1)  {  // Edge 4
00391         array_D.resize(Nx_D + 1);
00392         xya = shift[1] + overlap[1] + gy_D;
00393         dh = lx / (Nx_D - 1);
00394         for(int i = 1; i <= Nx_D; ++i)
00395             array_D(i) = (i - 1) * dh + shift[0];
00396         array_D(Nx_D + 1) = xya; // the y-coordinate
00397     }
00398 
00399     return 0;
00400 }
